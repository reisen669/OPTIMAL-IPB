# download_multires.py
import os
import sys

# Set PROJ_DATA before importing osgeo
os.environ['PROJ_DATA'] = r'C:\Users\suily\miniconda3\envs\qgis_gdal_env\Library\share\proj'

from osgeo import gdal
gdal.UseExceptions()

# ── Bounding box (EPSG:3857) from canvas_0.5mpx_extent.txt ──
ULX = 11464086.564073   # West
ULY = 377242.052866     # North
LRX = 11464728.435127   # East
LRY = 376763.883391     # South

OUTPUT_DIR = r'C:\Users\suily\AppData\Roaming\QGIS\QGIS3\profiles\default\python\plugins\optimal-ipb\sample_data_qgis\mys_lat3388N_lon10299E'
os.makedirs(OUTPUT_DIR, exist_ok=True)

# GDAL WMS XML template for ESRI World Imagery
# Note: ${{z}}, ${{y}}, ${{x}} become ${z}, ${y}, ${x} after .format() — required by GDAL TMS driver
WMS_TEMPLATE = """\
<GDAL_WMS>
  <Service name="TMS">
    <ServerUrl>https://services.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/${{z}}/${{y}}/${{x}}</ServerUrl>
  </Service>
  <DataWindow>
    <UpperLeftX>-20037508.34</UpperLeftX>
    <UpperLeftY>20037508.34</UpperLeftY>
    <LowerRightX>20037508.34</LowerRightX>
    <LowerRightY>-20037508.34</LowerRightY>
    <TileLevel>{zoom}</TileLevel>
    <TileCountX>1</TileCountX>
    <TileCountY>1</TileCountY>
    <YOrigin>top</YOrigin>
  </DataWindow>
  <Projection>EPSG:3857</Projection>
  <BlockSizeX>256</BlockSizeX>
  <BlockSizeY>256</BlockSizeY>
  <BandsCount>3</BandsCount>
  <MaxConnections>5</MaxConnections>
  <Cache/>
</GDAL_WMS>"""

# (target_res_m, tile_zoom, filename, fallback_zoom)
# fallback_zoom: if the primary zoom has no tiles, retry with this zoom
TARGETS = [
    (1.0, 17, 'imagery_1.0mpx.tif', None),
    (0.5, 18, 'imagery_0.5mpx.tif', None),
    (0.3, 19, 'imagery_0.3mpx.tif', 18),
    (0.1, 20, 'imagery_0.1mpx.tif', 19),
]

results = []

for target_res, zoom, filename, fallback_zoom in TARGETS:
    out_path = os.path.join(OUTPUT_DIR, filename)
    print(f'\n─── Downloading {filename} (target: {target_res} m/px, zoom: Z={zoom}) ───')

    used_zoom = zoom
    for attempt_zoom in ([zoom] + ([fallback_zoom] if fallback_zoom else [])):
        wms_xml = WMS_TEMPLATE.format(zoom=attempt_zoom)
        wms_file = os.path.join(OUTPUT_DIR, f'_tmp_esri_z{attempt_zoom}.xml')
        with open(wms_file, 'w') as f:
            f.write(wms_xml)

        try:
            warp_opts = gdal.WarpOptions(
                outputBounds=(ULX, LRY, LRX, ULY),
                outputBoundsSRS='EPSG:3857',
                dstSRS='EPSG:3857',
                xRes=target_res,
                yRes=target_res,
                resampleAlg='cubic',
                creationOptions=['COMPRESS=LZW', 'TILED=YES', 'BLOCKXSIZE=256', 'BLOCKYSIZE=256'],
                format='GTiff',
                multithread=True,
            )
            ds = gdal.Warp(out_path, wms_file, options=warp_opts)
            if ds is not None:
                w, h = ds.RasterXSize, ds.RasterYSize
                gt = ds.GetGeoTransform()
                native_res = abs(gt[1])
                ds = None
                used_zoom = attempt_zoom

                # Rename file if fallback was used (clarify source in filename)
                if used_zoom != zoom:
                    # Fallback was triggered — rename to indicate source
                    base_name = filename.replace('.tif', '')
                    new_filename = f'{base_name}_fallback_Z{used_zoom}.tif'
                    new_path = os.path.join(OUTPUT_DIR, new_filename)
                    os.rename(out_path, new_path)
                    filename = new_filename
                    print(f'  ✓ Created: {new_filename}  ({w} x {h} px, actual GSD: {native_res:.4f} m/px, zoom used: Z={attempt_zoom} [FALLBACK])')
                else:
                    print(f'  ✓ Created: {filename}  ({w} x {h} px, actual GSD: {native_res:.4f} m/px, zoom used: Z={attempt_zoom})')
                break
            else:
                print(f'  ✗ Z={attempt_zoom} returned None, trying fallback...')
        except Exception as e:
            print(f'  ✗ Z={attempt_zoom} failed: {e}')
        finally:
            if os.path.exists(wms_file):
                os.remove(wms_file)
    else:
        print(f'  ✗ FAILED: could not download {filename} at any zoom level')
        w, h, native_res = 0, 0, 0.0

    # Record result for summary
    size_mb = os.path.getsize(out_path) / 1_048_576 if os.path.exists(out_path) else 0
    results.append({
        'filename': filename,
        'target_res': target_res,
        'zoom': used_zoom,
        'width': w,
        'height': h,
        'actual_res': native_res,
        'size_mb': size_mb,
    })

print('\nAll downloads complete.')
print(results)
