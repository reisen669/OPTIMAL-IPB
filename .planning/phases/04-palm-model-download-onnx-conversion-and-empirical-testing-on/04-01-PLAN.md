```python
"""Verify all ONNX models in models/ load correctly with onnxruntime."""
import onnxruntime as ort
import os
import glob

plugin_dir = os.path.dirname(os.path.abspath(__file__))
models_dir = os.path.join(plugin_dir, 'models')

# Collect all ONNX files recursively (covers models/ root AND models/mopad/)
onnx_files = sorted(
    os.path.relpath(p, plugin_dir)
    for p in glob.glob(os.path.join(models_dir, '**', '*.onnx'), recursive=True)
)

print(f"Models directory: {models_dir}")
print(f"ONNX files found: {onnx_files}\n")

all_ok = True
for rel_path in onnx_files:
    path = os.path.join(plugin_dir, rel_path)
    size_mb = os.path.getsize(path) / 1024 / 1024
    try:
        sess = ort.InferenceSession(path, providers=['CPUExecutionProvider'])
        inputs  = [i.name for i in sess.get_inputs()]
        outputs = [o.name for o in sess.get_outputs()]
        print(f"  [OK]  {rel_path} ({size_mb:.0f} MB)")
        print(f"        inputs={inputs}")
        print(f"        outputs={outputs}")
    except Exception as e:
        print(f"  [FAIL] {rel_path} ({size_mb:.0f} MB): {e}")
        all_ok = False

print()
if all_ok:
    print(f"All {len(onnx_files)} ONNX models verified successfully.")
else:
    print("Some models failed verification.")
```

Key changes from original:
1. Added `import glob` at top
2. Added `plugin_dir` variable (dirname of the script itself)

4. Loop variable is now `rel_path` (relative to plugin_dir) instead of `fname`
5. Inner path construction is `os.path.join(plugin_dir, rel_path)` instead of `os.path.join(models_dir, fname)`
6. Print uses `rel_path` so output shows `models/mopad/MOPAD_epoch_24.onnx` (not just filename)
